261  NDARRAY_ASSERT(input);
262 #ifndef _MSC_VER
263  output = static_cast<T>(PyLong_AsUnsignedLongLong(input.get()));
264 #else
265  // Christoph Lassner reports that the above segfaults on Windows; until we have a better idea

32  static bool fromPythonStage1(PyPtr & p) {
33  // add or remove dimensions with size one so we have the right number of dimensions
34  if (PyArray_Check(p.get())) {
35  if ((Rows_ == 1 || Cols_ == 1) && N == 2) {
36  npy_intp shape[2] = { -1, -1 };
37  if (Rows_ == 1) {
38  shape[0] = 1;
39  } else {
40  shape[1] = 1;
41  }
42  PyArray_Dims dims = { shape, 2 };
43  PyPtr r(PyArray_Newshape(reinterpret_cast<PyArrayObject*>(p.get()), &dims, NPY_ANYORDER));
44  if (!r) return false;
45  p.swap(r);
46  } else if (N == 1) {
47  PyPtr r(PyArray_Squeeze(reinterpret_cast<PyArrayObject*>(p.get())));
48  if (!r) return false;
49  p.swap(r);
50  }
51  } // else let the Array converter raise the exception
52  if (!PyConverter< Array<T,N,C> >::fromPythonStage1(p)) return false;
53  // check whether the size is correct if it's static
54  auto array = reinterpret_cast<PyArrayObject*>(p.get());
55  if (N == 2) {
56  if (Rows_ != Eigen::Dynamic && PyArray_DIM(array, 0) != Rows_) {
57  PyErr_SetString(PyExc_ValueError, "incorrect number of rows for matrix");
58  return false;
59  }
60  if (Cols_ != Eigen::Dynamic && PyArray_DIM(array, 1) != Cols_) {
61  PyErr_SetString(PyExc_ValueError, "incorrect number of columns for matrix");
62  return false;
63  }
64  } else {
65  int requiredSize = Rows_ * Cols_;
66  if (requiredSize != Eigen::Dynamic && PyArray_SIZE(array) != requiredSize) {
67  PyErr_SetString(PyExc_ValueError, "incorrect number of elements for vector");
68  return false;
69  }
70  }
71  return true;
72  }

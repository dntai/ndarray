31  static boost::numpy::ndarray apply(Input const & input, bool writeable=true) {
32  boost::numpy::dtype dtype = boost::numpy::dtype::get_builtin<Scalar>();
33  int itemsize = dtype.get_itemsize();
34  std::vector<Py_intptr_t> shape(2);
35  shape[0] = input.rows();
36  shape[1] = input.cols();
37  std::vector<Py_intptr_t> strides(2, itemsize);
38  strides[0] *= input.rowStride();
39  strides[1] *= input.colStride();
40  boost::python::object none;
41  boost::numpy::ndarray result = (writeable) ?
42  boost::numpy::from_data(const_cast<Scalar const*>(input.data()), dtype, shape, strides, none) :
43  boost::numpy::from_data(const_cast<Scalar*>(input.data()), dtype, shape, strides, none);
44  if (Input::RowsAtCompileTime == 1 || Input::ColsAtCompileTime == 1) return result.squeeze();
45  return result;
46  }

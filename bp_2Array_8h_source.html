53  static boost::numpy::ndarray apply(Array<T,N,C> const & array) {
54  boost::numpy::dtype dtype
55  = boost::numpy::dtype::get_builtin<typename boost::remove_const<T>::type>();
56  boost::python::object owner = detail::makePyObject(array.getManager());
57  Py_ssize_t itemsize = dtype.get_itemsize();
58  ndarray::Vector<Size,N> shape_elements = array.getShape();
59  ndarray::Vector<Offset,N> strides_elements = array.getStrides();
60  std::vector<Py_intptr_t> shape_bytes(N);
61  std::vector<Py_intptr_t> strides_bytes(N);
62  for (int n=0; n<N; ++n) {
63  shape_bytes[n] = shape_elements[n];
64  strides_bytes[n] = strides_elements[n] * itemsize;
65  }
66  return boost::numpy::from_data(array.getData(), dtype, shape_bytes, strides_bytes, owner);
67  }

71 template <typename T, int N, int C>
72 class FromBoostPython< Array<T,N,C> > {
73 public:
74 
75  explicit FromBoostPython(boost::python::object const & input_) : input(input_) {}
76 
77  bool convertible() {
78  if (input.is_none()) return true;
79  try {
80  boost::numpy::ndarray array = boost::python::extract<boost::numpy::ndarray>(input);
81  boost::numpy::dtype dtype
82  = boost::numpy::dtype::get_builtin<typename boost::remove_const<T>::type>();
83  boost::numpy::ndarray::bitflag flags = array.get_flags();
84  if (dtype != array.get_dtype()) return false;
85  if (N != array.get_nd()) return false;
86  if (!boost::is_const<T>::value && !(flags & boost::numpy::ndarray::WRITEABLE)) return false;
87  if (C > 0) {
88  Offset requiredStride = sizeof(T);
89  for (int i = 0; i < C; ++i) {
90  if ((array.shape(N-i-1) > 1) && (array.strides(N-i-1) != requiredStride)) {
91  return false;
92  }
93  requiredStride *= array.shape(N-i-1);
94  }
95  } else if (C < 0) {
96  Offset requiredStride = sizeof(T);
97  for (int i = 0; i < -C; ++i) {
98  if ((array.shape(i) > 1) && (array.strides(i) != requiredStride)) {
99  return false;
100  }
101  requiredStride *= array.shape(i);
102  }
103  }
104  } catch (boost::python::error_already_set) {
105  boost::python::handle_exception();
106  PyErr_Clear();
107  return false;
108  }
109  return true;
110  }
111 
112  Array<T,N,C> operator()() {
113  if (input.is_none()) return Array<T,N,C>();
114  boost::numpy::ndarray array = boost::python::extract<boost::numpy::ndarray>(input);
115  boost::numpy::dtype dtype = array.get_dtype();
116  Py_ssize_t itemsize = dtype.get_itemsize();
117  for (int i = 0; i < N; ++i) {
118  if ((array.shape(i) > 1) && (array.strides(i) % itemsize != 0)) {
119  PyErr_SetString(
120  PyExc_TypeError,
121  "Cannot convert array to C++: strides must be an integer multiple of the element size"
122  );
123  boost::python::throw_error_already_set();
124  }
125  }
126  boost::python::object obj_owner = array.get_base();
127  if (obj_owner.is_none()) {
128  obj_owner = array;
129  }
130  Vector<Size,N> shape;
131  Vector<Offset,N> strides;
132  for (int i=0; i<N; ++i) {
133  shape[i] = array.shape(i);
134  if (shape[i] > 1) {
135  strides[i] = array.strides(i) / itemsize;
136  } else {
137  strides[i] = 1;
138  }
139  }
140  Array<T,N,C> r = ndarray::external(
141  reinterpret_cast<T*>(array.get_data()), shape, strides, obj_owner
142  );
143  return r;
144  }
145 
146  boost::python::object input;
147 };

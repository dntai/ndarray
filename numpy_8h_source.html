152  static bool fromPythonStage1(
153  PyPtr & p
157  ) {
158  if (!PyArray_Check(p.get())) {
159  PyErr_SetString(PyExc_TypeError, "numpy.ndarray argument required");
160  return false;
161  }
162  auto array = reinterpret_cast<PyArrayObject*>(p.get());
163  int actualType = PyArray_TYPE(array);
164  int requiredType = detail::NumpyTraits<NonConst>::getCode();
165  if (actualType != requiredType) {
166  PyErr_SetString(PyExc_ValueError, ("numpy.ndarray argument has incorrect data type"));
167  return false;
168  }
169  if (PyArray_NDIM(array) != N) {
170  PyErr_SetString(PyExc_ValueError, "numpy.ndarray argument has incorrect number of dimensions");
171  return false;
172  }
173  bool writeable = !boost::is_const<Element>::value;
174  if (writeable && !(PyArray_FLAGS(array) & NPY_ARRAY_WRITEABLE)) {
175  PyErr_SetString(PyExc_TypeError, "numpy.ndarray argument must be writeable");
176  return false;
177  }
178  if (C > 0) {
179  Offset requiredStride = sizeof(Element);
180  for (int i = 0; i < C; ++i) {
181  Offset actualStride = PyArray_STRIDE(array, N-i-1);
182  if (actualStride != requiredStride) {
183  PyErr_SetString(
184  PyExc_ValueError,
185  "numpy.ndarray does not have enough row-major contiguous dimensions"
186  );
187  return false;
188  }
189  requiredStride *= PyArray_DIM(array, N-i-1);
190  }
191  } else if (C < 0) {
192  int requiredStride = sizeof(Element);
193  for (int i = 0; i < -C; ++i) {
194  Offset actualStride = PyArray_STRIDE(array, i);
195  if (actualStride != requiredStride) {
196  PyErr_SetString(
197  PyExc_ValueError,
198  "numpy.ndarray does not have enough column-major contiguous dimensions"
199  );
200  return false;
201  }
202  requiredStride *= PyArray_DIM(array, i);
203  }
204  }
205  return true;
206  }

220  static bool fromPythonStage2(
221  PyPtr const & input,
222  Array<T,N,C> & output
223  ) {
224  auto array = reinterpret_cast<PyArrayObject*>(input.get());
225  if (!(PyArray_FLAGS(array) & NPY_ARRAY_ALIGNED)) {
226  PyErr_SetString(PyExc_TypeError, "unaligned arrays cannot be converted to C++");
227  return false;
228  }
229  Offset itemsize = sizeof(Element);
230  for (int i = 0; i < N; ++i) {
231  if ((PyArray_DIM(array, i) > 1) && (PyArray_STRIDE(array, i) % itemsize != 0)) {
232  PyErr_SetString(
233  PyExc_TypeError,
234  "Cannot convert array to C++: strides must be an integer multiple of the element size"
235  );
236  return false;
237  }
238  }
239  Vector<Size,N> shape;
240  Vector<Offset,N> strides;
241  std::copy(PyArray_DIMS(array), PyArray_DIMS(array) + N, shape.begin());
242  std::copy(PyArray_STRIDES(array), PyArray_STRIDES(array) + N , strides.begin());
243  for (int i = 0; i < N; ++i) strides[i] /= sizeof(Element);
244  output = external(
245  reinterpret_cast<Element*>(PyArray_DATA(array)),
246  shape, strides, input
247  );
248  return true;
249  }

261  static PyObject* toPython(
262  Array<T,N,C> const & m,
263  PyObject* owner=NULL
265  ) {
266  int flags = NPY_ARRAY_ALIGNED;
267  if (C==N) flags |= NPY_ARRAY_C_CONTIGUOUS;
268  bool writeable = !boost::is_const<Element>::value;
269  if (writeable) flags |= NPY_ARRAY_WRITEABLE;
270  npy_intp outShape[N];
271  npy_intp outStrides[N];
272  Vector<Size,N> inShape = m.getShape();
273  Vector<Offset,N> inStrides = m.getStrides();
274  std::copy(inShape.begin(), inShape.end(), outShape);
275  for (int i = 0; i < N; ++i) outStrides[i] = inStrides[i] * sizeof(Element);
276  PyPtr array(
277  PyArray_New(
278  &PyArray_Type, N, outShape, detail::NumpyTraits<NonConst>::getCode(),
279  outStrides, const_cast<NonConst*>(m.getData()), sizeof(Element), flags, NULL
280  ),
281  false
282  );
283  if (!array) return NULL;
284  if (!m.getManager() && owner == NULL) {
285  flags = NPY_ARRAY_CARRAY_RO | NPY_ARRAY_ENSURECOPY | NPY_ARRAY_C_CONTIGUOUS;
286  if (writeable) flags |= NPY_ARRAY_WRITEABLE;
287  PyPtr r = PyArray_FROM_OF(array.get(),flags);
288  if (!r) return NULL;
289  array.swap(r);
290  } else {
291  if (owner != NULL) {
292  Py_INCREF(owner);
293  } else {
294  owner = PyCapsule_New(
295  new Manager::Ptr(m.getManager()),
296  "ndarray.Manager",
297  detail::destroyCapsule
298  );
299  }
300  PyArray_SetBaseObject(reinterpret_cast<PyArrayObject*>(array.get()), owner);
301  }
302  Py_INCREF(array.get());
303  return PyArray_Return(reinterpret_cast<PyArrayObject*>(array.get()));
304  }
